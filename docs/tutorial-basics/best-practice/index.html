Redis Stream Consumer Boilerplate

After a few projects I‚Äôve settled on a code idiom that improves development time and covers the most common requirements:

  • Always listen for new messages from a Redis Stream.
  • Always reclaim ‚Äústuck‚Äù messages that sit in the pending list too long.

üí° Tip: if your app uses scheduled recovery, don‚Äôt forget @EnableScheduling on your Spring Boot application class.

Step 1: Abstract base class‚Äã

Create an abstract base class to hold common behaviour:

  • Initialise the Redis consumer group
  • Register the main consumer loop with @EventListener(ApplicationReadyEvent.class)
  • Add a shutdown hook for a clean stop
  • Expose abstract map() and handle() so concrete consumers focus only on business logic
@Slf4j
public abstract class AbstractBaseConsumer<T> {

    protected final StringRedisTemplate redisTemplate;
    protected final String streamKey;
    protected final String groupName;
    protected final String consumerName;
    private volatile boolean running = true;

    protected final ExecutorService executor = new ThreadPoolExecutor(
        2, 4, 60, TimeUnit.SECONDS,
        new LinkedBlockingQueue<>(100),
        new ThreadPoolExecutor.AbortPolicy()
    );

    protected AbstractBaseConsumer(StringRedisTemplate redisTemplate,
                                   String streamKey,
                                   String groupName,
                                   String consumerName) {
        this.redisTemplate = redisTemplate;
        this.streamKey = streamKey;
        this.groupName = groupName;
        this.consumerName = consumerName;
    }

    @PostConstruct
    public void initGroup() {
        try {
            redisTemplate.opsForStream().createGroup(streamKey, ReadOffset.from("0"), groupName);
            log.info("Created group '{}' on stream '{}'", groupName, streamKey);
        } catch (RedisSystemException e) {
            if (e.getCause() instanceof io.lettuce.core.RedisBusyException) {
                log.info("Group '{}' already exists", groupName);
            } else {
                throw e;
            }
        }
    }

    @PreDestroy
    public void shutdown() {
        running = false;
        executor.shutdownNow();
    }

    @EventListener(ApplicationReadyEvent.class)
    public void start() {
        executor.submit(this::loop);
    }

    private void loop() {
        while (running) {
            try {
                List<MapRecord<String,Object,Object>> records = redisTemplate.opsForStream().read(
                    Consumer.from(groupName, consumerName),
                    StreamReadOptions.empty().block(Duration.ofMinutes(1)).count(10),
                    StreamOffset.create(streamKey, ReadOffset.lastConsumed())
                );

                if (records == null || records.isEmpty()) continue;

                processBatch(records);
            } catch (Exception e) {
                log.error("Stream processing error", e);
                try {
                    Thread.sleep(1000);
                } catch (InterruptedException ie) {
                    Thread.currentThread().interrupt();
                    break;
                }
            }
        }
        log.info("Consumer {} stopped.", consumerName);
    }

    protected void processBatch(List<MapRecord<String,Object,Object>> records) {
        for (var record : records) {
            try {
                T payload = map(record.getValue());
                handle(payload);
                ack(record);
            } catch (Exception e) {
                onError(record.getId().getValue(), record, e);
            }
        }
    }

    protected void ack(MapRecord<String,Object,Object> record) {
        redisTemplate.opsForStream().acknowledge(streamKey, groupName, record.getId());
    }

    protected void onError(String streamId, MapRecord<String,Object,Object> record, Exception e) {
        log.error("Processing failed for streamId: {}", streamId, e);
        // Optionally: publish to DLQ + ack on mapping/validation errors
    }

    protected abstract T map(Map<Object, Object> fields);
    protected abstract void handle(T payload) throws Exception;
}

MapStruct‚Äã

For payload mapping, use MapStruct to convert the Redis map into a Java class.

Eg: Given this payload model:

@Data
@Builder
@NoArgsConstructor
@AllArgsConstructor
public class RaceData {
    private String rider;
    private String speed;
    private String position;
    private String locationId;
}

Create a mapper

import org.mapstruct.Mapper;
import org.mapstruct.Mapping;
import org.mapstruct.NullValueCheckStrategy;
import org.mapstruct.ReportingPolicy;

import java.util.Map;

@Mapper(
    componentModel = "spring",
    nullValueCheckStrategy = NullValueCheckStrategy.ALWAYS,
    unmappedTargetPolicy = ReportingPolicy.IFNORE
)
public interface RaceDataMapper {

    @Mapping(target = "rider",     expression = "java(asString(map.get(\"rider\")))")
    @Mapping(target = "speed",     expression = "java(asString(map.get(\"speed\")))")
    @Mapping(target = "position",  expression = "java(asString(map.get(\"position\")))")
    @Mapping(target = "locationId",expression = "java(asString(map.get(\"location_id\")))")
    RaceData fromMap(Map<Object, Object> map);

    default String asString(Object v) {
        return v == null ? null : String.valueOf(v).trim();
    }
}

Consumer class‚Äã

A concrete consumer wires in the business service and the mapper, and implements map()/handle().


@Slf4j
@Component
public class RaceFranceConsumer extends AbstractBaseConsumer<RaceData> {

    private final YourBusinessService service;
    private final RaceDataMapper mapper;

    public RaceFranceConsumer(StringRedisTemplate redisTemplate,
                              YourBusinessService service,
                              RaceDataMapper mapper,
                              @Qualifier("hostname") String consumerName) {
        super(redisTemplate,
              RedisEventTypes.RACE_STREAM_KEY,
              RedisEventTypes.RACE_GROUP_NAME,
              consumerName);
        this.service = service;
        this.mapper = mapper;
    }

    @Override
    protected RaceData map(Map<Object,Object> fields) {
        return mapper.fromMap(fields);
    }

    @Override
    protected void handle(RaceData payload) throws Exception {
        service.process(payload);
    }
}

Stuck message recovery‚Äã

Recovery consumers also extend the base class, and add a @Scheduled task to reclaim idle messages.

(Keep the idle timeout longer than your worst-case processing time unless you also implement heartbeats.)

@Slf4j
@Component
public class RaceFranceRecovery extends AbstractBaseConsumer<RaceData> {

    private final YourBusinessService service;
    private final RaceDataMapper mapper;
    private final String CONSUMER_NAME;

    @Value("${stream.idle-timeout-minutes:15}")
    private long idleTimeoutMinutes;

    public RaceFranceRecovery(YourBusinessService service,
                              RaceDataMapper mapper,
                              StringRedisTemplate redisTemplate,
                              @Qualifier("hostname") String consumerName) {
        super(redisTemplate,
              RedisEventTypes.RACE_STREAM_KEY,
              RedisEventTypes.RACE_GROUP_NAME,
              consumerName + ":recovery");
        this.service = service;
        this.mapper = mapper;
        this.CONSUMER_NAME = consumerName;
    }

    @Override
    public void start() {
        // do nothing ‚Äî prevents loop
    }

    @Override
    protected RaceData map(Map<Object,Object> fields) {
        return mapper.fromMap(fields);
    }

    @Override
    protected void handle(RaceData payload) throws Exception {
        service.process(payload);
    }

    @Scheduled(fixedRate = 600_000) // Depends your business logic
    public void reclaimStuckMessages() {
        log.info("Running job to reclaim stuck messages...");
        try {
            PendingMessagesSummary summary = redisTemplate.opsForStream()
                .pending(RedisEventTypes.RACE_STREAM_KEY, RedisEventTypes.RACE_GROUP_NAME);

            if (summary == null || summary.getTotalPendingMessages() == 0) return;

            log.info("Found {} total pending messages. Checking for idle ones...",
                     summary.getTotalPendingMessages());

            PendingMessages pending = redisTemplate.opsForStream().pending(
                RedisEventTypes.RACE_STREAM_KEY,
                RedisEventTypes.RACE_GROUP_NAME,
                Range.unbounded(),
                100L
            );

            for (PendingMessage pm : pending) {
                if (pm.getElapsedTimeSinceLastDelivery()
                      .compareTo(Duration.ofMinutes(idleTimeoutMinutes)) > 0) {
                    log.warn("Message {} by '{}' idle for {} ‚Üí reclaiming",
                             pm.getId(), pm.getConsumerName(), pm.getElapsedTimeSinceLastDelivery());

                    List<MapRecord<String, Object, Object>> claimed = redisTemplate.opsForStream().claim(
                        RedisEventTypes.RACE_STREAM_KEY,
                        RedisEventTypes.RACE_GROUP_NAME,
                        CONSUMER_NAME + ":recovery",
                        Duration.ofSeconds(30),
                        pm.getId()
                    );

                    if (!claimed.isEmpty()) {
                        executor.submit(() -> processBatch(claimed));
                    }
                }
            }
        } catch (DataAccessException e) {
            log.error("Redis access error during message reclamation: {}", e.getMessage(), e);
        } catch (Exception e) {
            log.error("Unexpected error during message reclamation", e);
        }
    }
}

Optional but recommended

  • DLQ + ACK for poison messages (mapping/validation errors) so they don‚Äôt loop forever.
  • Max retry budget: after N deliveries (from XPENDING), move to DLQ and XACK.
  • XAUTOCLAIM instead of XCLAIM for batch reclaiming of idle messages.
  • Heartbeat to avoid reclaiming messages that are long-running but healthy.

With this pattern the boilerplate lives in the base class, consumers and recovery jobs stay lean, and you can focus entirely on business logic.
